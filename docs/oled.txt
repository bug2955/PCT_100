#include "oled_display.h"

// ★ 已修改：使用硬件 I2C (HW_I2C)，极大降低 CPU 占用，屏幕刷新如丝般顺滑
U8G2_SH1106_128X64_VCOMH0_F_HW_I2C u8g2(U8G2_R0, U8X8_PIN_NONE);

void oled_init(void) {
    // 将底层硬件 I2C 高速通道映射到引脚 4 (SDA) 和 5 (SCL)
    Wire.begin(I2C_SDA, I2C_SCL); 
    u8g2.begin();
    u8g2.enableUTF8Print(); // 开启 UTF8 支持打印中文
}

void oled_update(String mode, bool masterOn, uint16_t light, uint16_t lightThresh, float temp, float tempThresh, bool lightOn, bool fanOn) {
    u8g2.clearBuffer(); // 清空显存

    // 设置中文字体 (文泉驿14号字体)
    u8g2.setFont(u8g2_font_wqy14_t_gb2312); 

    // ------------------------------------------------
    // 第一行：模式与总闸 (左右分栏对齐)
    // ------------------------------------------------
    u8g2.setCursor(0, 14);
    u8g2.print("模式: ");
    u8g2.print(mode == "auto" ? "自动" : "手动");
    
    u8g2.setCursor(70, 14); // 强制在 X=70 像素处对齐第二列
    u8g2.print("总闸: ");
    u8g2.print(masterOn ? "ON" : "OFF");

    // ------------------------------------------------
    // 第二行：光照数据
    // ------------------------------------------------
    u8g2.setCursor(0, 30);
    u8g2.print("光照: ");
    u8g2.print(light);
    u8g2.print(" / ");
    u8g2.print(lightThresh);

    // ------------------------------------------------
    // 第三行：温度数据
    // ------------------------------------------------
    u8g2.setCursor(0, 46);
    u8g2.print("温度: ");
    u8g2.print(temp, 1); // 保留1位小数
    u8g2.print(" / ");
    u8g2.print(tempThresh, 1);

    // ------------------------------------------------
    // 第四行：继电器状态 (左右分栏对齐)
    // ------------------------------------------------
    u8g2.setCursor(0, 62);
    u8g2.print("灯光: ");
    u8g2.print(lightOn ? "ON" : "OFF");

    u8g2.setCursor(70, 62); // 与第一行的总闸对齐
    u8g2.print("风扇: ");
    u8g2.print(fanOn ? "ON" : "OFF");

    u8g2.sendBuffer(); // 将显存内容一次性推送到屏幕，防止闪烁
}






#ifndef __OLED_DISPLAY_H
#define __OLED_DISPLAY_H

#include <Arduino.h>
#include <U8g2lib.h>
#include <Wire.h>

// ★ 参考老师给的代码截图，设定引脚
#define I2C_SDA 4
#define I2C_SCL 5

// 函数声明
void oled_init(void);
// 统一更新屏幕数据的接口
void oled_update(String mode, bool masterOn, uint16_t light, uint16_t lightThresh, float temp, float tempThresh, bool lightOn, bool fanOn);

#endif
