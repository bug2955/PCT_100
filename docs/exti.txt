#include "exti.h"

// 中断标志（volatile 必须加！）
static volatile bool keyFlag = false;
static unsigned long lastDebounce = 0;

// 中断服务函数（极简！！！）
void IRAM_ATTR keyISR() {
  unsigned long now = millis();
  if (now - lastDebounce > DEBOUNCE_MS) {
    keyFlag = true;
    lastDebounce = now;
  }
}

// 初始化按键 + 中断
void keyInit(void) {
  pinMode(KEY_PIN, INPUT_PULLUP);
  attachInterrupt(KEY_PIN, keyISR, KEY_PRESS_TRIGGER);
}

// 读取按键
bool keyPressed(void) {
  return keyFlag;
}

// 清除标志
void keyClear(void) {
  keyFlag = false;
}





#ifndef exti_h
#define exti_h

#include <Arduino.h>

// ==================== ESP32-C3 配置 ====================
#define KEY_PIN             0        // 按键引脚（可任意修改：0~22）
#define DEBOUNCE_MS         20       // 消抖时间 20ms
#define KEY_PRESS_TRIGGER   FALLING  // 上拉模式：按下 = 下降沿

// 初始化函数
void keyInit(void);

// 获取按键触发标志
bool keyPressed(void);

// 清除标志
void keyClear(void);

#endif
